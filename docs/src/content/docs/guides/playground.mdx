## Full demo

Variable-width slides with drag, navigation, pagination, and autoplay.

<Playground
  id="ex-featured"
  label="Featured items"
  plugins="drag,navigation,pagination,autoplay"
  slides={['Variable width A', 'Slide B', 'Wider slide C', 'Slide D']}
  code={`import { Unswipe } from 'unswipe';
import { autoplay } from 'unswipe/plugins/autoplay';
import { drag } from 'unswipe/plugins/drag';
import { navigation } from 'unswipe/plugins/navigation';
import { pagination } from 'unswipe/plugins/pagination';

new Unswipe(root, { align: 'start', label: 'Featured items' }, [
  drag(),
  navigation(),
  pagination(),
  autoplay({ delay: 4500 }),
]);`}
>
  <div slot="controls" class="playground__controls">
    <button type="button" data-action="add" data-target="ex-featured">
      Add slide
    </button>
    <button type="button" data-action="remove" data-target="ex-featured">
      Remove last
    </button>
  </div>
</Playground>
