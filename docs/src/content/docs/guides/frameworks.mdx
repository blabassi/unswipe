    ```js
    class UnswipeCarousel extends HTMLElement {
      #slider;

      connectedCallback() {
        this.classList.add('carousel');
        this.#slider = new Unswipe(this, {
          label: this.getAttribute('label') ?? 'Carousel',
          align: 'start',
        });
      }

      disconnectedCallback() {
        this.#slider?.destroy();
      }
    }

    customElements.define('unswipe-carousel', UnswipeCarousel);
    ```

    ```tsx
    import { useEffect, useRef } from 'react';
    import { Unswipe } from 'unswipe';

    export function Carousel({ slides, label = 'Carousel' }) {
      const ref = useRef(null);

      useEffect(() => {
        const root = ref.current;
        if (!root) return;
        const slider = new Unswipe(root, { label, align: 'start' });
        return () => slider.destroy();
      }, [label]);

      return (
        <div ref={ref} className="carousel">
          {slides.map((text) => (
            <div key={text} className="slide">
              {text}
            </div>
          ))}
        </div>
      );
    }
    ```

    ```vue
    <script setup>
    import { onMounted, onUnmounted, ref } from 'vue';
    import { Unswipe } from 'unswipe';

    const root = ref(null);
    let slider;

    onMounted(() => {
      slider = new Unswipe(root.value, { label: 'Products' });
    });

    onUnmounted(() => slider?.destroy());
    </script>

    <template>
      <div ref="root" class="carousel">
        <div v-for="n in 3" :key="n" class="slide">Slide {{ n }}</div>
      </div>
    </template>
    ```
