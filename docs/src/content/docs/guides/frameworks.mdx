Unswipe is framework-agnostic. Use the official thin wrappers below, or mount on any DOM node and call `destroy()` on teardown.

<Tabs>
  <TabItem label="React">

```tsx
import { useUnswipe } from 'unswipe/react';
import { drag } from 'unswipe/plugins/drag';
import 'unswipe/style.css';

export function Carousel({ slides }: { slides: string[] }) {
  const { ref } = useUnswipe({ label: 'Products' }, [drag()]);

  return (
    <div ref={ref} className="carousel">
      {slides.map((text) => (
        <div key={text} className="slide">
          {text}
        </div>
      ))}
    </div>
  );
}
```

```vue
<script setup>
import { useUnswipe } from 'unswipe/vue';
import { drag } from 'unswipe/plugins/drag';
import 'unswipe/style.css';

const { el } = useUnswipe({ label: 'Products' }, [drag()]);
</script>

<template>
  <div ref="el" class="carousel">
    <div v-for="n in 3" :key="n" class="slide">Slide {{ n }}</div>
  </div>
</template>
```

```svelte
<script>
  import { unswipe } from 'unswipe/svelte';
  import { drag } from 'unswipe/plugins/drag';
  import 'unswipe/style.css';
</script>

<div class="carousel" use:unswipe={{ options: { label: 'Gallery' }, plugins: [drag()] }}>
  {#each ['A', 'B', 'C'] as label}
    <div class="slide">{label}</div>
  {/each}
</div>
```

```js
class UnswipeCarousel extends HTMLElement {
  #slider;

  connectedCallback() {
    this.classList.add('carousel');
    // Defer if this can mount inside a hidden tab — % slide widths need a real box.
    this.#slider = new Unswipe(this, {
      label: this.getAttribute('label') ?? 'Carousel',
    });
  }

  disconnectedCallback() {
    this.#slider?.destroy();
  }
}

customElements.define('unswipe-carousel', UnswipeCarousel);
```
